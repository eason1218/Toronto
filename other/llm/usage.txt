library(ggplot2)
library(readr)

# Read the data from the CSV file
file_path <- 'path_to_your_file/analysis_data.csv'
df <- read_csv(file_path)

# Convert the 'OCCUPANCY_DATE' column to Date type
df$OCCUPANCY_DATE <- as.Date(df$OCCUPANCY_DATE, format="%Y-%m-%d")

# Convert the occupancy rates to numeric
df$OCCUPANCY_RATE_BEDS <- as.numeric(df$OCCUPANCY_RATE_BEDS)
df$OCCUPANCY_RATE_ROOMS <- as.numeric(df$OCCUPANCY_RATE_ROOMS)

# Filter out rows with NA occupancy rates
occupancy_data <- df[!is.na(df$OCCUPANCY_RATE_BEDS) | !is.na(df$OCCUPANCY_RATE_ROOMS), ]

# Create a line plot for bed and room occupancy rates
ggplot(occupancy_data, aes(x=OCCUPANCY_DATE)) +
  geom_line(aes(y=OCCUPANCY_RATE_BEDS, color="Bed Occupancy Rate"), na.rm=TRUE) +
  geom_line(aes(y=OCCUPANCY_RATE_ROOMS, color="Room Occupancy Rate"), na.rm=TRUE) +
  labs(title="Bed and Room Occupancy Rates Over Time", 
       x="Date", y="Occupancy Rate (%)") +
  scale_color_manual(values=c("Bed Occupancy Rate"="blue", "Room Occupancy Rate"="green")) +
  theme_minimal() +
  theme(axis.text.x = element_text(angle=45, hjust=1))
